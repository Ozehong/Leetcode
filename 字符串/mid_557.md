class Solution {
public:
    void reverse(string &s,int i,int j)
    {
        for(;i<j;i++,j--)
            swap(s[i],s[j]);
    }
    string reverseWords(string s) {
        int n=s.length();
        int left=0;
        for(int i=0;i<n;i++)
        {
            while(i<n && s[i]!=' ')
                i++;  
            reverse(s,left,i++-1);
            left=i;
        }
        return s;
    }
};
