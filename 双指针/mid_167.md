# 方法1：第一个按顺序，第二个二分查询，复杂度nlogn
class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
        int n=numbers.size();
        vector<int> ans(2,0);
        for(int i=0;i<n;i++)
        {
            int left=i+1,right=n-1;
            while(left<=right)
            {
                int mid=(right-left)/2+left;
                if(numbers[i]+numbers[mid]==target)
                {
                    ans[0]=i+1;
                    ans[1]=mid+1;
                    return ans;
                }
                else
                {
                    if(numbers[i]+numbers[mid]>target)
                        right=mid-1;
                    else left=mid+1;
                }
            }
        }
        return ans;
    }
};
